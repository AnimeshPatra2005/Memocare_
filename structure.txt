memocare/ (Root Directory)
├── .env.example                # Example configuration template (ports, DB keys, service URLs)
├── .gitignore                  # Git ignore rules (ignores virtual envs, node_modules, and model weights)
├── docker-compose.yml          # Orchestrates all containers (Postgres, Gateway, Services, Frontend)
├── README.md                   # Quickstart guide for running the application
│
├── gateway/                    # Single entry point for the frontend
│   ├── Dockerfile              # Docker config for building the Gateway container
│   ├── requirements.txt        # gateway dependencies (fastapi, uvicorn, httpx, jose)
│   └── main.py                 # Routes incoming HTTP requests to corresponding backend services
│
├── frontend/                   # React/Vite SPA (Move your existing frontend files here)
│   ├── ...                     
│   └── src/
│       └── utils/
│           └── crypto.js       # Client-side AES-GCM encryption helper using Web Crypto API
│
└── services/                   # Folder grouping all backend microservices
    │
    ├── auth/                   # Authentication Service
    │   ├── Dockerfile          # Docker config for Auth container
    │   ├── requirements.txt    # Python packages (fastapi, passlib[bcrypt], pyjwt, sqlalchemy)
    │   ├── database.py         # Handles SQLAlchemy DB engine and Session setup
    │   ├── models.py           # SQL model for the 'users' table (stores email, password hash, and salt)
    │   └── main.py             # Endpoints for registering, logging in, and retrieving salt
    │
    ├── db/                     # Medical Records DB Service (Zero-Knowledge)
    │   ├── Dockerfile          # Docker config for DB service container
    │   ├── requirements.txt    # Python packages (fastapi, sqlalchemy, psycopg2-binary)
    │   ├── database.py         # Handles SQLAlchemy DB connection for records
    │   ├── models.py           # SQL model for the 'medical_records' table (stores E2E encrypted JSON and IVs)
    │   └── main.py             # CRUD endpoints for saving/getting user-specific encrypted records
    │
    ├── model_mri/              # Model 1 Service (MRI Scans)
    │   ├── Dockerfile          # Docker config (runs Python + Keras/TensorFlow dependencies)
    │   ├── requirements.txt    # ML packages (tensorflow, numpy, pillow, fastapi)
    │   ├── main.py             # Loads DenseNet201 + custom weights and exposes a prediction endpoint
    │   └── mri_models/         # Local storage for Model 1 weights (e.g. densenet_weights.h5)
    │
    ├── model_tabular/          # Model 2 Service (Ensemble tab-clinical data)
    │   ├── Dockerfile          # Docker config (runs Scikit-Learn packages)
    │   ├── requirements.txt    # Packages (scikit-learn, joblib, pandas, numpy, fastapi)
    │   ├── main.py             # Loads scaler and stacking classifier, returns predictions
    │   └── models/             # Local storage for model binaries (scaler.pkl and stacking_model.pkl)
    │
    └── pdf_reports/            # Document Generator Service
        ├── Dockerfile          # Docker config
        ├── requirements.txt    # Packages (reportlab, fastapi)
        └── main.py             # Takes plaintext prediction details and outputs a binary PDF report
